To restart the Gunicorn server:
# 1) See who is listening on 8001
sudo ss -ltnp | grep ':8001' || true

# 2) Stop BOTH gunicorn services (important)
sudo systemctl stop ip_transcoder-v2_gunicorn.service
sudo systemctl stop ip_transcoder_gunicorn.service

# 3) Check again - if something still listens on 8001, kill it
sudo ss -ltnp | grep ':8001' || true
# If you still see a PID like pid=XXXXX, kill it:
# sudo kill -9 XXXXX

# 4) Disable the OLD service so it never comes back (keep only v2)
sudo systemctl disable --now ip_transcoder_gunicorn.service

# 5) Reload systemd (because you edited unit files)
sudo systemctl daemon-reload

# 6) Start ONLY v2
sudo systemctl start ip_transcoder-v2_gunicorn.service
sudo systemctl status ip_transcoder-v2_gunicorn.service -l --no-pager
